Installation d'OCS Inventory NG

sudo apt install apache2 mariadb-server php libapache2-mod-perl2 ...
wget https://github.com/OCSInventory-NG/.../OCSNG_UNIX_SERVER.tar.gz
tar xvzf OCSNG_UNIX_SERVER.tar.gz
sudo ./setup.sh

Installation de GLPI

wget https://github.com/glpi-project/glpi/releases/download/...
tar xvzf glpi-10.0.3.tgz -C /var/www/html
sudo chown -R www-data:www-data /var/www/html/glpi
